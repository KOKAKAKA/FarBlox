			local baller

			if training_mode then
				baller = getTrainingBall()
			else
				baller = parryfunct.shiscripbykia()
			end

			if not baller then
				weirddname.hit_Time = tick() * 10
				weirddname.hit_Count = 0
				weirddname.is_ball_Warping = false
				weirddname.notimportant = false
				weirddname.balball = true
				weirddname.last_target = nil

				return

			end

			baller:GetAttributeChangedSignal('target'):Once(function()
				weirddname.balball = true
			end)

			if baller:GetAttribute('target') ~= local_player.Name or not weirddname.balball then
				return
			end

			baller:GetAttributeChangedSignal('from'):Once(function()
				weirddname.last_target = workspace.Alive:FindFirstChild(baller:GetAttribute('from'))
			end)

			closestplayer(local_player.Character.PrimaryPart)

			local player_Position = local_player.Character.PrimaryPart.Position

			local ball_Position = baller.Position

			local ball_Velocity = baller.AssemblyLinearVelocity

			if baller:FindFirstChild("zoomies") then

				ball_Velocity = baller.zoomies.VectorVelocity

			end

			local ball_Direction = (local_player.Character.PrimaryPart.Position - ball_Position).Unit

			local ball_Distance = (local_player.Character.PrimaryPart.Position - ball_Position).Magnitude

			local ball_Dot = ball_Direction:Dot(ball_Velocity.Unit)

			local ball_Speed = ball_Velocity.Magnitude

			if ball_Speed < 25 then
				weirddname.hit_Time = tick() * 10
				weirddname.hit_Count = 0
				weirddname.is_ball_Warping = false
				weirddname.notimportant = false
				weirddname.can_parry = true
				weirddname.last_target = nil
			end

			local player_Position = local_player.Character.PrimaryPart.Position
			local player_Velocity = local_player.Character.HumanoidRootPart.AssemblyLinearVelocity
			local player_isMoving = player_Velocity.Magnitude > 0

			local ball_speed_Limited = math.min(ball_Speed / 1000, 0.1)

			local ball_predicted_Distance = (ball_Distance - ping / 15.5) - (ball_Speed / 3.5)

			local target_Position = closest_Entity.HumanoidRootPart.Position

			local target_Distance = local_player:DistanceFromCharacter(target_Position)

			local closest_target_Distance = local_player:DistanceFromCharacter(closest_Entity.HumanoidRootPart.Position)

			local target_distance_Limited = math.min(target_Distance / 10000, 0.1)

			local target_Direction = (local_player.Character.PrimaryPart.Position - closest_Entity.HumanoidRootPart.Position).Unit

			local target_Velocity = closest_Entity.HumanoidRootPart.AssemblyLinearVelocity

			local target_isMoving = target_Velocity.Magnitude > 0

			local target_Dot = target_isMoving and math.max(target_Direction:Dot(target_Velocity.Unit), 0)

			weirddname.spam_Range = math.clamp(math.max(math.max(ping / 10, 15) + ball_Speed / 7.5, 8.4), 30, 1000)

			local speedFactor = 3.3 - (ping / 100)

			local speedFactorPingBased = 5.3 - (ping / 100)

			weirddname.autoparrylol = math.max(math.max(ping, 3.5) + ball_Speed / speedFactor, speedFactorPingBased)

			if target_isMoving then

				weirddname.notimportant = weirddname.hit_Count > 1 or (target_Distance <= weirddname.spam_Range and ball_Distance < 10 and ball_Speed > 10) and ball_Velocity.Y > -50 and ball_Dot > 0.2

			else

				weirddname.notimportant = weirddname.hit_Count > 1 or (target_Distance <= weirddname.spam_Range and ball_Distance < 10 and ball_Speed > 10) and ball_Velocity.Y > -50 and ball_Dot > 0.2
			end

			local velocity_angle = math.abs(math.acos(ball_Direction:Dot(ball_Velocity.Unit)))

			if ball_Dot < -0.2 then

				weirddname.ball_Warping = tick()

			end

			task.spawn(function()

				if (tick() - weirddname.ball_Warping) >= 0.15 + target_distance_Limited - ball_speed_Limited or ball_Distance <= 10 then

					weirddname.is_ball_Warping = false

					return

				end

				if weirddname.last_target then
					if (ball_Position - weirddname.last_target.HumanoidRootPart.Position).Magnitude > 35.5 or target_Distance <= 12 then
						weirddname.is_ball_Warping = false
					end
				end

				weirddname.is_ball_Warping = true

			end)

			if ball_Velocity.Y < -150 and baller:FindFirstChild("AeroDynamicSlash") then
				task.wait(0.1)
			end

			if ball_Distance <= weirddname.autoparrylol and not weirddname.is_ball_Warping and ball_Speed > 0 and ball_Dot > 0.1 then

				if auto_curve then
					thisismath:FireServer(
						0,
						CFrame.new(camera.CFrame.Position, Vector3.new(math.random(-1000, 1000), math.random(0, 1000), math.random(100, 1000))),
						{[closest_Entity_To_mouse.Name] = closest_Entity_To_mouse.HumanoidRootPart.Position},
						{closest_Entity_To_mouse.HumanoidRootPart.Position.X, closest_Entity_To_mouse.HumanoidRootPart.Position.Y},
						false
					)
				else
					local cf = camera.CFrame
					local x, y, z, R00, R01, R02, R10, R11, R12, R20, R21, R22 = cf:GetComponents()
			
					thisismath:FireServer(
						0,
						CFrame.new(x, y, z, R00, R01, R02, R10, R11, R12, R20, R21, R22),
						{[closest_Entity_To_mouse.Name] = closest_Entity_To_mouse.HumanoidRootPart.Position},
						{closest_Entity_To_mouse.HumanoidRootPart.Position.X, closest_Entity_To_mouse.HumanoidRootPart.Position.Y},
						false
					)
				end

				weirddname.hit_Count = weirddname.hit_Count + 1
				task.delay(0.55 + (ping / 100), function()
					weirddname.hit_Count = weirddname.hit_Count - 1
				end)

				weirddname.hit_Time = tick()

				weirddname.balball = false

			end

			task.spawn(function()

				repeat

					RunService.PreRender:Wait()

				until (tick() - weirddname.hit_Time) >= 1

				weirddname.balball = true

			end)

		end)

	end)
